        async function saveSubAdmin(event) {
            event.preventDefault();
            
            // Enforce frontend branch limit: max 11 unique branches
            try {
                const checkRes = await fetch('/api/admin/sub-admins', {
                    headers: { 'Authorization': `Bearer ${localStorage.getItem('sapthala_token')}` }
                });
                if (checkRes.ok) {
                    const existing = await checkRes.json();
                    const branches = new Set(existing.map(s => s.branch).filter(Boolean));
                    const editId = document.getElementById('subAdminEditId').value;
                    const newBranch = document.getElementById('subAdminBranch').value;
                    // If creating new (not editing) and branch is new and limit reached, block
                    if (!editId && !branches.has(newBranch) && branches.size >= 11) {
                        alert('Branch limit reached. You cannot create more than 11 branches.');
                        return;
                    }
                }
            } catch (e) {
                console.warn('Branch limit check failed (continuing):', e);
            }

            const permissions = {};
            document.querySelectorAll('input[name="permissions"]:checked').forEach(cb => {
                permissions[cb.value] = true;
            });

            const formData = {
                username: document.getElementById('subAdminUsername').value,
                password: document.getElementById('subAdminPassword').value,
                branch: document.getElementById('subAdminBranch').value,
                permissions
            };

            // Normalize branch input: support "Display (ID)" or plain name
            try {
                let raw = (formData.branch || '').toString().trim();
                if (!raw) {
                    // allow server to handle empty
                } else {
                    const m = raw.match(/\(([^)]+)\)\s*$/);
                    if (m && m[1]) {
                        formData.branch = m[1].trim();
                    } else if (window.branchReverseMap && window.branchReverseMap[raw.toLowerCase()]) {
                        formData.branch = window.branchReverseMap[raw.toLowerCase()];
                    } else {
                        formData.branch = raw;
                    }
                }
            } catch (e) { console.warn('Branch normalization failed', e); }

            try {
                const response = await fetch('/api/admin/sub-admins', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${localStorage.getItem('sapthala_token')}`
                    },
                    body: JSON.stringify(formData)
                });
                
                if (!response.ok) {
                    // try to read error message
                    let errMsg = 'Failed to save sub-admin';
                    try {
                        const err = await response.json();
                        errMsg = err.error || err.message || errMsg;
                    } catch (e) {
                        console.warn('Non-JSON error response', e);
                    }
                    alert('Failed to save sub-admin: ' + errMsg);
                    // If server says only main admin can create, provide clearer UX
                    return;
                }

                alert('Sub-admin created successfully!');
                closeSubAdminModal();
                await loadSubAdmins();
                // Refresh branch lists across the UI so staff/selects/reports reflect the new branch
                try { await refreshBranchesAndPopulate(); } catch (e) { console.warn('refreshBranches failed after create', e); }
                try { await loadStaff(); } catch (e) { console.warn('loadStaff failed after create', e); }
                try { await loadReports(); } catch (e) { console.warn('loadReports failed after create', e); }
            } catch (error) {
                console.error('saveSubAdmin error', error);
                alert('Failed to save sub-admin: ' + (error.message || error));
            }
        }

                const aNum = parseInt(a.staffId.replace('staff_', '')) || 999;

            });

                            ${member.name.charAt(0).toUpperCase()}

        async function editStaff(staffId) {
            try {
                console.log(' Editing staff:', staffId);
                
                const response = await fetch('/api/staff');
                if (!response.ok) throw new Error('Failed to fetch staff');
                
                const staff = await response.json();
                // Try to find the staff member by multiple identifier fields returned by different API shapes
                const member = staff.find(s => s._id === staffId || s.id === staffId || s.staffId === staffId || (s._id && s._id.toString() === staffId));
                
                if (!member) throw new Error('Staff member not found');

                document.getElementById('staffModalTitle').textContent = ' Edit Staff';
                // For updates we need a Mongo _id where possible; prefer member._id then member.id, otherwise pass the provided id (may be staffId)
                document.getElementById('staffEditId').value = member._id || member.id || staffId;
                document.getElementById('staffUsername').value = member.staffId;
                document.getElementById('staffName').value = member.name;
                document.getElementById('staffPhone').value = member.phone;
                document.getElementById('staffEmail').value = member.email || '';
                document.getElementById('staffRole').value = member.role;
                document.getElementById('staffPin').value = member.pin;
                // member.branch is stored as branchId; ensure we set select value accordingly
                document.getElementById('staffBranch').value = member.branch || '';
                
                document.querySelectorAll('input[name="stages"]').forEach(cb => {
                    cb.checked = member.workflowStages.includes(cb.value);
                });
                
                document.getElementById('staffModal').classList.add('show');
                
            } catch (error) {
                console.error(' Edit staff failed:', error);
                alert(' Failed to load staff details: ' + error.message);
            }
        }

        // Prompt admin to ensure a branch has staff for all workflow stages
        function ensureBranchStaffPrompt() {
            // Prefer the branch selected in the staff filter select
            const filter = document.getElementById('staffFilterBranch');
            let branch = filter && filter.value ? filter.value : null;
            if (!branch) {
                branch = prompt('Enter branchId or branchName to ensure staff for (e.g. SAPTHALA.MAIN or Main):');
            }
            if (!branch) return;
            ensureBranchStaff(branch);
        }

        async function ensureBranchStaff(branch) {
            try {
                const token = localStorage.getItem('sapthala_token');
                if (!token) {
                    alert(' You must be logged in as admin to perform this action');
                    return;
                }
                const res = await fetch('/api/admin/ensure-branch-staff', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` },
                    body: JSON.stringify({ branch })
                });
                if (!res.ok) {
                    const err = await res.json().catch(()=>({ error: 'Failed' }));
                    alert(' Could not ensure branch staff: ' + (err.error || err.message || 'Server error'));
                    return;
                }
                const data = await res.json();

                // Try to fetch branch display name for friendlier messages
                let branchDisplay = branch;
                try {
                    const bres = await fetch('/api/public/branches');
                    if (bres && bres.ok) {
                        const blist = await bres.json();
                        const found = (blist || []).find(b => (b.branchId && b.branchId.toString().toLowerCase() === branch.toString().toLowerCase()) || (b.branchName && b.branchName.toString().toLowerCase() === branch.toString().toLowerCase()));
                        if (found) branchDisplay = found.branchName || found.branchId || branch;
                    }
                } catch (e) {
                    console.warn('Could not fetch branches for display name', e);
                }

                const created = Array.isArray(data.created) ? data.created : [];
                if (created.length > 0) {
                    // Refresh staff list and navigate to staff tab
                    await loadStaff();
                    showTab('staff');

                    // Inform user and open edit modal for the first created staff (best-effort)
                    showInfoModal('Branch Staffing', ` Created ${created.length} staff member(s) for ${branchDisplay}. Opening staff list...`);

                    // Try to open the first created staff in the edit modal
                    const first = created[0];
                    const firstId = first && (first._id || first.id || first.staffId);
                    // Resolve to Mongo _id if necessary
                    const resolved = firstId ? await resolveStaffMongoId(firstId) : null;
                    setTimeout(async () => {
                        try {
                            // Ensure staff list is fresh
                            await loadStaff();
                            // Open edit modal for the created staff (if resolvable)
                            if (resolved || firstId) {
                                editStaff(resolved || firstId);
                            }
                        } catch (e) {
                            console.warn('Could not auto-open created staff modal', e);
                        }
                    }, 600);
                } else {
                    alert(' No new staff needed for branch ' + branchDisplay);
                    await loadStaff();
                }
            } catch (e) {
                console.error('ensureBranchStaff error', e);
                alert(' Failed to ensure branch staff: ' + e.message);
            }
        }

        async function deleteStaff(staffId) {

                const resolvedId = await resolveStaffMongoId(staffId);

            } catch (error) {

            }
        }

        // Helper: try to map different API identifier shapes to Mongo _id by fetching staff list
        async function resolveStaffMongoId(identifier) {
            try {
                if (!identifier) return null;
                // Quick check for ObjectId pattern
                if (/^[0-9a-fA-F]{24}$/.test(identifier)) return identifier;
                const res = await fetch('/api/staff');
                if (!res.ok) return null;
                const list = await res.json();
                const member = list.find(s => s._id === identifier || s.id === identifier || s.staffId === identifier || (s._id && s._id.toString() === identifier));
                if (member) return member._id || member.id || null;
            } catch (e) {
                console.warn('resolveStaffMongoId failed', e);
            }
            return null;
        }

        // Helper: refresh branch list (public/auth) and populate branch selects/datalists across the UI
        async function refreshBranchesAndPopulate() {
            try {
                const token = localStorage.getItem('sapthala_token') || sessionStorage.getItem('sapthala_token');
                let branches = [];
                if (token) {
                    try {
                        const res = await fetch('/api/branches', { headers: { 'Authorization': `Bearer ${token}` } });
                        if (res.ok) {
                            const data = await res.json();
                            branches = Array.isArray(data.branches) ? data.branches : (Array.isArray(data) ? data : []);
                        }
                    } catch (e) { /* ignore and fallback */ }
                }
                if (!branches || branches.length === 0) {
                    try {
                        const pub = await fetch('/api/public/branches');
                        if (pub.ok) branches = await pub.json();
                    } catch (e) { /* ignore */ }
                }

                // Build mappings for friendly display names and reverse lookup
                window.branchMap = window.branchMap || {};
                window.branchReverseMap = window.branchReverseMap || {};
                (branches || []).forEach(b => {
                    const id = (b.branchId || b.branch || b.branchName || '').toString();
                    const name = (b.branchName || b.branchId || b.branch || '').toString();
                    if (id) {
                        window.branchMap[id] = name;
                        window.branchReverseMap[name.toLowerCase()] = id;
                        // also map id lowercased for flexible reverse lookup
                        window.branchReverseMap[id.toLowerCase()] = id;
                    }
                });

                // Populate UI selects/datalists
                const datalist = document.getElementById('branchesDatalist');
                const staffBranchSelect = document.getElementById('staffBranch');
                const staffFilter = document.getElementById('staffFilterBranch');
                const reportBranch = document.getElementById('reportBranch');
                const orderBranch = document.getElementById('orderBranch');

                if (datalist) datalist.innerHTML = '';
                if (staffBranchSelect) staffBranchSelect.innerHTML = '<option value="">Select branch</option>';
                if (staffFilter) staffFilter.innerHTML = '<option value="">All Branches</option>';
                if (reportBranch) reportBranch.innerHTML = '<option value="">All Branches</option>';
                if (orderBranch) orderBranch.innerHTML = '<option value="">Choose branch...</option>';

                (branches || []).forEach(b => {
                    const display = (b.branchName || b.branchId || b.branch || '').toString();
                    const value = (b.branchId || display).toString();
                    if (datalist) {
                        const opt = document.createElement('option'); opt.value = display; datalist.appendChild(opt);
                    }
                    if (staffBranchSelect) {
                        const opt = document.createElement('option'); opt.value = value; opt.textContent = `${display} (${value})`; staffBranchSelect.appendChild(opt);
                    }
                    if (staffFilter) {
                        const o2 = document.createElement('option'); o2.value = value; o2.textContent = `${display} (${value})`; staffFilter.appendChild(o2);
                    }
                    if (reportBranch) {
                        const o3 = document.createElement('option'); o3.value = value; o3.textContent = `${display} (${value})`; reportBranch.appendChild(o3);
                    }
                    if (orderBranch) {
                        const o4 = document.createElement('option'); o4.value = value; o4.textContent = `${display} (${value})`; orderBranch.appendChild(o4);
                    }
                });
            } catch (e) {
                console.warn('refreshBranchesAndPopulate failed', e);
            }
        }

                const orderData = collectOrderData();

                }

                alert(' Failed to generate PDF: ' + error.message);

        }

        // NEW FUNCTION: Apply UI restrictions based on user role
        function applyPermissionRestrictions() {
            try {
                const userStr = localStorage.getItem('sapthala_user') || sessionStorage.getItem('sapthala_user');
                if (!userStr) return;
                const user = JSON.parse(userStr);

                // If sub-admin, apply restrictions
                if (user.role === 'sub-admin') {
                    console.log(' Applying sub-admin restrictions...');

                    // Hide Sub-Admins tab (sub-admin should not manage other sub-admins)
                    const subAdminsTab = document.getElementById('subAdminsTab');
                    if (subAdminsTab) subAdminsTab.style.display = 'none';

                    // Hide Staff tab if no permission to manage staff
                    if (!user.permissions?.canManageStaff) {
                        const staffTab = Array.from(document.querySelectorAll('.sidebar-tab')).find(t => t.textContent && t.textContent.includes('Staff'));
                        if (staffTab) staffTab.style.display = 'none';
                    }

                    // Hide Reports tab if no permission
                    if (!user.permissions?.canViewReports) {
                        const reportsTab = Array.from(document.querySelectorAll('.sidebar-tab')).find(t => t.textContent && t.textContent.includes('Reports'));
                        if (reportsTab) reportsTab.style.display = 'none';
                    }

                    // Hide revenue cards on dashboard
                    hideRevenueData();

                    // Hide revenue and category charts on dashboard
                    try {
                        const revenueCanvas = document.getElementById('revenueChart');
                        if (revenueCanvas && revenueCanvas.parentElement) revenueCanvas.parentElement.style.display = 'none';
                        const categoryCanvas = document.getElementById('categoryChart');
                        if (categoryCanvas && categoryCanvas.parentElement) categoryCanvas.parentElement.style.display = 'none';
                    } catch (e) {
                        console.warn('Could not hide charts for sub-admin:', e);
                    }

                    // Hide Reports panel and download buttons
                    try {
                        const reportsPanel = document.getElementById('reports');
                        if (reportsPanel) reportsPanel.style.display = 'none';

                        const downloadPdfBtn = Array.from(document.querySelectorAll('button')).find(b => b.getAttribute('onclick') && b.getAttribute('onclick').includes('downloadReportPDF'));
                        if (downloadPdfBtn) downloadPdfBtn.style.display = 'none';
                    } catch (e) {
                        console.warn('Could not hide reports for sub-admin:', e);
                    }

                    // Gray-out theme cards but keep visible
                    try {
                        document.querySelectorAll('.theme-card').forEach(tc => tc.classList.add('disabled'));
                        const applyBtn = Array.from(document.querySelectorAll('button')).find(b => b.getAttribute('onclick') && b.getAttribute('onclick').includes('applyTheme'));
                        if (applyBtn) applyBtn.disabled = true, applyBtn.style.opacity = '0.6', applyBtn.style.pointerEvents = 'none';
                    } catch (e) {
                        console.warn('Could not disable themes for sub-admin:', e);
                    }

                    console.log(' Sub-admin restrictions applied');
                }
            } catch (error) {
                console.error('Error applying permissions:', error);
            }
        }

        // NEW FUNCTION: Hide revenue-related data for sub-admins
        function hideRevenueData() {
            // Hide Total Revenue card
            const revenueCard = document.getElementById('totalRevenue');
            if (revenueCard && revenueCard.parentElement) {

            // Hide Advance Collected card

            }
        }
